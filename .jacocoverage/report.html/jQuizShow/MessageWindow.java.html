        m_text.setDisabledTextColor(java.awt.Color.black);
        m_text.setEnabled(false);
        jScrollPane1.setViewportView(m_text);
        
        getContentPane().add(jScrollPane1, java.awt.BorderLayout.CENTER);
        
        pack();
